```python id="c4t8bn"
import random

user_wins = 0
com_wins = 0

dic_rps_ref = {
    '가위': {'가위': None, '바위': '컴','보': '사'},
    '바위': {'가위': '사', '바위': None,'보': '컴'},
    '보': {'가위': '컴', '바위':'사','보': None},
}
# Define computer's choices as strings for comparison with user_rps
list_com_choices = ['가위', '바위', '보']

print("가위바위보 게임을 시작합니다! 먼저 3승을 하는 사람이 이깁니다.")

while user_wins < 3 and com_wins < 3: # Continue until one player wins 3 rounds
    user_rps = None
    while True: # Loop for valid user input
        user_rps = input("가위, 바위, 보 중에서 하나를 선택/입력해 주세요 : ")
        if user_rps in ['가위','바위','보']:
            print(f"사용자가 선택한 것은 {user_rps} 입니다.")
            break
        print("사용자가 잘못 선택했어요. 다시 입력해 주세요.")

    # Computer makes a choice
    com_rps_index = random.randint(0, 2)
    com_rps = list_com_choices[com_rps_index]
    print(f"컴퓨터가 선택한 것은 {com_rps} 입니다.")

    # Determine the winner of the current round
    winner = dic_rps_ref[user_rps][com_rps]

    if winner == '사':
        user_wins += 1
        print("이번 라운드는 사용자가 이겼습니다!")
    elif winner == '컴':
        com_wins += 1
        print("이번 라운드는 컴퓨터가 이겼습니다!")
    else: # winner is None (draw)
        print("이번 라운드는 비겼습니다!")

    print(f"현재 점수: 사용자 {user_wins} : 컴퓨터 {com_wins}")
    print("-" * 30) # Separator for rounds

# After the game ends (one player reached 3 wins)
if user_wins == 3:
    print("축하합니다! 사용자가 최종 승리했습니다! 🎉")
else:
    print("아쉽지만 컴퓨터가 최종 승리했습니다. 🤖")

```

* switch문 구조 이해
* break를 통한 탈출
* case 값 설정 방식 학습

👉 다양한 조건 처리 방식 비교 이해
