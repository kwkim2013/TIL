#### 1. 클래스(Class)와 객체(Object)

- 클래스는 객체를 생성하기 위한 설계도
- 객체는 클래스를 기반으로 만들어진 실제 데이터
- 하나의 클래스로 여러 개의 객체를 생성할 수 있음

예:
```python
class Calculator:
    pass

calc1 = Calculator()
calc2 = Calculator()
```


#### ✔ 2. 생성자 __init__와 self
- __init__는 객체가 생성될 때 자동으로 실행되는 함수
- self는 자기 자신(객체)을 의미
- 객체의 속성(변수)을 초기화할 때 사용

```python
class PhoneBook:

    def __init__(self):
        self.contacts = []
```

설명:

**self.contacts**
→ 이 객체만 사용하는 연락처 리스트 생성
→ 객체가 만들어질 때 자동으로 빈 리스트가 준비됨

#### 3. 계산기 클래스 구현
- 사칙연산 기능을 메서드로 분리하여 구현
- 입력값을 받아 계산 결과를 반환하도록 구성


```python
예:

class Calculator:

    def add(self, a, b):
        return a + b

    def sub(self, a, b):
        return a - b

    def mul(self, a, b):
        return a * b

    def div(self, a, b):
        if b == 0:
            return "0으로 나눌 수 없습니다."
        return a / b
```

#### ✔ 4. 전화번호부 클래스 구현
연락처를 리스트로 저장
추가 / 삭제 / 조회 기능 구현

예:
```python
class PhoneBook:

    def __init__(self):
        self.contacts = []

    def add_contact(self, name, phone):
        self.contacts.append({
            "name": name,
            "phone": phone
        })

    def show_contacts(self):
        for contact in self.contacts:
            print(contact["name"], contact["phone"])

    def delete_contact(self, name):
        for contact in self.contacts:
            if contact["name"] == name:
                self.contacts.remove(contact)
                print("삭제 완료")
```

#### 5. 상속 + 딕셔너리 + keys() + items() 포함 완성형 코드
```python
class ContactBookBase:

    def __init__(self):
        # 기본 클래스의 초기화
        print("ContactBookBase가 초기화되었습니다.")


class PhoneBook(ContactBookBase):  # ContactBookBase로부터 상속

    # 함수 초기화
    def __init__(self):
        super().__init__()  # 부모 클래스 생성자 호출
        self.contacts = {}  # 연락처를 딕셔너리로 변경: 이름 -> 전화번호

    # 연락처 추가
    def add_contact(self, name, phone):
        self.contacts[name] = phone  # 이름을 키(key)로 사용
        print(f"'{name}' 연락처가 추가되었습니다.")

    # 전체 연락처 출력
    def show_all(self):
        if not self.contacts:
            print("연락처가 비어 있습니다!")
            return

        print("----- 모든 연락처 -----")

        for name, phone in self.contacts.items():  # items() 사용
            print(f"이름: {name}, 전화번호: {phone}")

        print("------------------------")

    # 이름으로 검색
    def search_by_name(self):
        name = input("검색할 이름을 입력하세요: ")

        if name in self.contacts.keys():  # keys() 사용
            print("전화번호:", self.contacts[name])
            return

        print(f"'{name}'을(를) 찾을 수 없습니다!")

    # 연락처 삭제
    def delete_contact(self):
        name = input("삭제할 이름을 입력하세요: ")

        if name in self.contacts:
            del self.contacts[name]
            print(f"'{name}' 연락처가 삭제되었습니다.")
            return

        print(f"'{name}'을(를) 찾을 수 없습니다!")

    # 이름순 정렬 출력 (key() 개념 연결)
    def sort_contacts(self):
        if not self.contacts:
            print("연락처가 없습니다.")
            return

        print("----- 이름순 정렬 -----")

        for name in sorted(self.contacts.keys()):  # key 기준 정렬
            print(f"이름: {name}, 전화번호: {self.contacts[name]}")

        print("------------------------")

```

#### 6. CSV 파일 저장 / 불러오기
```python
import csv

class ContactBookBase:

    def __init__(self):
        print("ContactBookBase가 초기화되었습니다.")

class PhoneBook(ContactBookBase):

    def __init__(self):
        super().__init__()
        self.contacts = {}

    # 연락처 추가
    def add_contact(self, name, phone):
        try:
            if not name.strip() or not phone.strip():
                raise ValueError("이름과 전화번호는 비워둘 수 없습니다.")

            self.contacts[name] = phone
            print(f"'{name}' 연락처가 추가되었습니다.")

        except ValueError as e:
            print("입력 오류:", e)

    # 전체 연락처 출력
    def show_all(self):
        try:
            if not self.contacts:
                print("연락처가 비어 있습니다!")
                return

            print("----- 모든 연락처 -----")
            for name, phone in self.contacts.items():
                print(f"이름: {name}, 전화번호: {phone}")
            print("------------------------")

        except Exception as e:
            print("출력 중 오류 발생:", e)

    # 이름으로 검색
    def search_by_name(self):
        try:
            name = input("검색할 이름을 입력하세요: ").strip()

            if not name:
                raise ValueError("이름을 입력해야 합니다.")

            if name in self.contacts.keys():
                print("전화번호:", self.contacts[name])
            else:
                print(f"'{name}'을(를) 찾을 수 없습니다!")

        except ValueError as e:
            print("입력 오류:", e)

    # 연락처 삭제
    def delete_contact(self):
        try:
            name = input("삭제할 이름을 입력하세요: ").strip()

            if not name:
                raise ValueError("이름을 입력해야 합니다.")

            if name in self.contacts:
                del self.contacts[name]
                print(f"'{name}' 연락처가 삭제되었습니다.")
            else:
                print(f"'{name}'을(를) 찾을 수 없습니다!")

        except ValueError as e:
            print("입력 오류:", e)

    # 이름순 정렬 출력
    def sort_contacts(self):
        try:
            if not self.contacts:
                print("연락처가 없습니다.")
                return

            print("----- 이름순 정렬 -----")
            for name in sorted(self.contacts.keys()):
                print(f"이름: {name}, 전화번호: {self.contacts[name]}")
            print("------------------------")

        except Exception as e:
            print("정렬 중 오류 발생:", e)

    # CSV 파일 저장
    def save_to_csv(self, filename="contacts.csv"):
        try:
            with open(filename, mode="w", newline="", encoding="utf-8-sig") as file:
                writer = csv.writer(file)
                writer.writerow(["이름", "전화번호"])  # 헤더

                for name, phone in self.contacts.items():
                    writer.writerow([name, phone])

            print(f"'{filename}' 파일로 저장되었습니다.")

        except Exception as e:
            print("CSV 저장 중 오류 발생:", e)

    # CSV 파일 불러오기
    def load_from_csv(self, filename="contacts.csv"):
        try:
            with open(filename, mode="r", newline="", encoding="utf-8-sig") as file:
                reader = csv.reader(file)
                next(reader)  # 헤더 건너뛰기

                self.contacts.clear()

                for row in reader:
                    if len(row) == 2:
                        name, phone = row
                        self.contacts[name] = phone

            print(f"'{filename}' 파일을 불러왔습니다.")

        except FileNotFoundError:
            print(f"'{filename}' 파일이 존재하지 않습니다.")

        except Exception as e:
            print("CSV 불러오기 중 오류 발생:", e)

```

| **init**   | 객체 생성 시 자동 실행되는 함수  |

### 💡 느낀 점 / 이해

이번 수업을 통해 클래스(Class)를 활용하여 계산기와 전화번호부 프로그램을 직접 구현해보면서  
객체지향 프로그래밍이 단순한 문법이 아니라 프로그램 구조를 체계적으로 만드는 방법이라는 것을 이해하게 되었다.  

특히 `__init__` 함수가 객체가 생성될 때 자동으로 실행되어 필요한 데이터를 초기화한다는 점과  
`self`가 현재 객체 자신을 의미하며, 객체의 속성과 메서드를 연결하는 역할을 한다는 것을 실습을 통해 체감할 수 있었다.  

전화번호부 프로그램을 구현하면서 리스트(List)와 딕셔너리(Dictionary)를 함께 사용하여  
여러 개의 데이터를 구조적으로 관리하는 방법을 이해할 수 있었고,  
기능을 메서드로 나누어 작성하면 코드의 재사용성과 가독성이 높아진다는 점도 확인할 수 있었다.  

또한 단순히 코드를 작성하는 것보다 프로그램의 흐름을 먼저 생각하고  
기능을 설계한 뒤 클래스로 구현하는 과정이 중요하다는 것을 느끼게 되었으며,  
앞으로는 문제를 해결할 때 절차적으로만 접근하기보다 객체 단위로 구조를 설계하는 습관을 가져야겠다고 생각했다.
